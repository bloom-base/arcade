   484→+                dead: false,
   485→+            };
   486→+        }

   488→+        // ── Start wave ────────────────────────────────────────

   504→+        }

   530→+        // ── Particle factory ─────────────────────────────────
   531→+        function addParticles(x, y, color, count = 6) {
   532→+            for (let i = 0; i < count; i++) {
   533→+                const angle = Math.random() * Math.PI * 2;
   534→+                const spd = 30 + Math.random() * 60;
   535→+                particles.push({
   536→+                    x, y,
   537→+                    vx: Math.cos(angle) * spd,
   538→+                    vy: Math.sin(angle) * spd,
   539→+                    life: 1.0,
   540→+                    color,
   541→+                    size: 2 + Math.random() * 3,
   542→+                });
   543→+            }
   544→+        }

   546→+        // ── Enemy movement ────────────────────────────────────
   547→+        function moveEnemy(e, dt) {
   548→+            if (e.slowTimer > 0) {
   549→+                e.slowTimer -= dt;
   550→+                e.speed = e.baseSpeed * (e.slowTimer > 0 ? e.slowFactor || e.baseSpeed : 1);
   551→+            } else {
   552→+                e.speed = e.baseSpeed;
   553→+            }
   554→+
   555→+            const [tx, ty] = WAYPOINTS[e.wpIdx];
   556→+            const dx = tx - e.x, dy = ty - e.y;
   557→+            const dist = Math.hypot(dx, dy);
   558→+            const move = e.speed * dt;
   559→+
   560→+            if (dist <= move) {
   561→+                e.x = tx; e.y = ty;
   562→+                e.wpIdx++;
   563→+                if (e.wpIdx >= WAYPOINTS.length) {
   564→+                    // Reached the end
   565→+                    e.dead = true;
   566→+                    lives = Math.max(0, lives - 1);
   567→+                    sfx.loseLife();
   568→+                    addFloat(GRID_W - 60, BAR_H / 2, '-1 LIFE', '#ef4444');
   569→+                    if (lives <= 0) endGame(false);
   570→+                }
   571→+            } else {
   572→+                e.x += (dx / dist) * move;
   573→+                e.y += (dy / dist) * move;
   574→+            }
   575→+        }
   576→+
   577→+        // ── Tower targeting ───────────────────────────────────
   578→+        function findTarget(tower) {
   579→+            // "First" targeting: enemy furthest along the path
   580→+            let best = null, bestProg = -1;
   581→+            for (const e of enemies) {
   582→+                if (e.dead) continue;
   583→+                const gx = e.x, gy = e.y;
   584→+                const dist = Math.hypot(gx - tower.x, gy - (tower.y));
   585→+                if (dist > tower.range) continue;
   586→+                const prog = e.wpIdx + (1 - Math.hypot(WAYPOINTS[e.wpIdx][0] - e.x, WAYPOINTS[e.wpIdx][1] - e.y) / 200);
   587→+                if (prog > bestProg) { bestProg = prog; best = e; }
   588→+            }
   589→+            return best;
   590→+        }

   592→+        // ── Tower firing ──────────────────────────────────────
   593→+        function fireTower(tower, dt) {
   594→+            tower.cooldown -= dt;
   595→+            if (tower.cooldown > 0) return;
   596→+            const target = findTarget(tower);
   597→+            if (!target) { tower.target = null; return; }
   598→+            tower.target = target;
   599→+            tower.angle = Math.atan2(target.y - tower.y, target.x - tower.x);
   600→+            tower.cooldown = 1 / tower.fireRate;
   601→+
   602→+            if (tower.type === 'lightning') {
   603→+                sfx.lightning();
   604→+                // Instant chain damage
   605→+                const hit = new Set([target]);
   606→+                target.hp -= tower.damage;
   607→+                addParticles(target.x, target.y + BAR_H, target.color);
   608→+                let last = target;
   609→+                for (let c = 1; c < tower.chainCount; c++) {
   610→+                    let next = null, bestD = Infinity;
   611→+                    for (const e of enemies) {
   612→+                        if (e.dead || hit.has(e)) continue;
   613→+                        const d = Math.hypot(e.x - last.x, e.y - last.y);
   614→+                        if (d < tower.chainRange && d < bestD) { bestD = d; next = e; }
   615→+                    }
   616→+                    if (!next) break;
   617→+                    hit.add(next);
   618→+                    next.hp -= tower.damage * 0.7;
   619→+                    addParticles(next.x, next.y + BAR_H, next.color);
   620→+                    // Lightning bolt visual via projectile stub
   621→+                    projectiles.push({
   622→+                        x: last.x, y: last.y,
   623→+                        tx: next.x, ty: next.y,
   624→+                        target: null, done: true,
   625→+                        visual: true, visualLife: 0.12,
   626→+                        color: tower.projColor, type: 'lightning',
   627→+                    });
   628→+                    last = next;
   629→+                }
   630→+                // Connector from tower to first target
   631→+                projectiles.push({
   632→+                    x: tower.x, y: tower.y,
   633→+                    tx: target.x, ty: target.y,
   634→+                    target: null, done: true,
   635→+                    visual: true, visualLife: 0.12,
   636→+                    color: tower.projColor, type: 'lightning',
   637→+                });
   638→+                // Kill check
   639→+                for (const e of hit) {
   640→+                    if (e.hp <= 0) killEnemy(e);
   641→+                }
   642→+            } else {
   643→+                const proj = createProjectile(tower, target);
   644→+                projectiles.push(proj);
   645→+                if (tower.type === 'cannon') sfx.cannon();
   646→+                else if (tower.type === 'single') sfx.shoot();
   647→+                else if (tower.type === 'slow') sfx.ice();
   648→+                else sfx.shoot();
   649→+            }
   650→+        }

   652→+        function killEnemy(e) {
   653→+            if (e.dead) return;
   654→+            e.dead = true;
   655→+            gold += e.reward;
   656→+            score += e.reward;
   657→+            sfx.kill();
   658→+            addParticles(e.x, e.y + BAR_H, e.color, 8);
   659→+            addFloat(e.x, e.y + BAR_H, '+' + e.reward, '#facc15');
   660→+        }

   662→+        // ── Projectile movement ───────────────────────────────
   663→+        function moveProjectile(p, dt) {
   664→+            if (p.visual) {
   665→+                p.visualLife -= dt;
   666→+                if (p.visualLife <= 0) p.done = true;
   667→+                return;
   668→+            }
   669→+            if (!p.target || p.target.dead) { p.done = true; return; }
   670→+            p.tx = p.target.x; p.ty = p.target.y;
   671→+            const dx = p.tx - p.x, dy = p.ty - p.y;
   672→+            const dist = Math.hypot(dx, dy);
   673→+            const move = p.speed * dt;
   674→+            if (dist <= move) {
   675→+                // Hit
   676→+                if (p.type === 'splash') {
   677→+                    for (const e of enemies) {
   678→+                        if (e.dead) continue;
   679→+                        const d = Math.hypot(e.x - p.tx, e.y - p.ty);
   680→+                        if (d <= p.splashRadius) {
   681→+                            e.hp -= p.damage * (1 - d / p.splashRadius * 0.5);
   682→+                            if (e.hp <= 0) killEnemy(e);
   683→+                        }
   684→+                    }
   685→+                    addParticles(p.tx, p.ty + BAR_H, '#fdba74', 12);
   686→+                } else {
   687→+                    p.target.hp -= p.damage;
   688→+                    if (p.type === 'slow') {
   689→+                        p.target.slowTimer = p.slowDuration;
   690→+                        p.target.slowFactor = p.slowFactor;
   691→+                    }
   692→+                    if (p.target.hp <= 0) killEnemy(p.target);
   693→+                    else addParticles(p.tx, p.ty + BAR_H, p.color, 3);
   694→+                }
   695→+                p.done = true;
   696→+            } else {
   697→+                p.x += (dx / dist) * move;
   698→+                p.y += (dy / dist) * move;
   699→+            }
   700→+        }

   873→+        // ── Update ────────────────────────────────────────────
   874→+        function update(dt) {
   875→+            if (gamePhase !== 'playing') return;
   876→+
   877→+            // Wave countdown
   878→+            if (!waveActive) {
   879→+                waveCountdown -= dt;
   880→+                if (waveCountdown <= 0) {
   881→+                    if (wave < TOTAL_WAVES) startWave();
   882→+                    else { endGame(true); return; }
   883→+                }
   884→+            }
   885→+
   886→+            // Spawn enemies
   887→+            if (waveActive && waveSpawnIdx < waveEnemies.length) {
   888→+                waveSpawnTimer += dt;
   889→+                while (waveSpawnIdx < waveEnemies.length &&
   890→+                       waveSpawnTimer >= waveEnemies[waveSpawnIdx].delay) {
   891→+                    enemies.push(createEnemy(waveEnemies[waveSpawnIdx].type));
   892→+                    waveSpawnIdx++;
   893→+                }
   894→+            }
   895→+
   896→+            // Move enemies
   897→+            for (const e of enemies) {
   898→+                if (!e.dead) moveEnemy(e, dt);
   899→+            }
   900→+
   901→+            // Tower logic
   902→+            for (const t of towers) fireTower(t, dt);
   903→+
   904→+            // Move projectiles
   905→+            for (const p of projectiles) moveProjectile(p, dt);
   906→+
   907→+            // Check wave complete
   908→+            if (waveActive && waveSpawnIdx >= waveEnemies.length && enemies.every(e => e.dead)) {
   909→+                waveActive = false;
   910→+                waveComplete = true;
   911→+                waveCountdown = 6;
   912→+                // Wave clear bonus
   913→+                const bonus = 20 + wave * 5;
   914→+                gold += bonus;
   915→+                score += bonus;
   916→+                addFloat(GRID_W / 2, BAR_H / 2, `WAVE ${wave} CLEAR! +${bonus}`, '#22c55e');
   917→+                if (wave >= TOTAL_WAVES) { setTimeout(() => endGame(true), 1200); }
   918→+            }
   919→+
   920→+            // Particles
   921→+            for (const p of particles) {
   922→+                p.x += p.vx * dt;
   923→+                p.y += p.vy * dt;
   924→+                p.vy += 80 * dt;
   925→+                p.life -= dt * 1.5;
   926→+            }
   927→+
   928→+            // Floats
   929→+            for (const f of floats) {
   930→+                f.y -= 30 * dt;
   931→+                f.life -= dt;
   932→+            }
   933→+
   934→+            // Cleanup
   935→+            enemies     = enemies.filter(e => !e.dead || e.wpIdx < WAYPOINTS.length);
   936→+            projectiles = projectiles.filter(p => !p.done);
   937→+            particles   = particles.filter(p => p.life > 0);
   938→+            floats      = floats.filter(f => f.life > 0);
   939→+        }
